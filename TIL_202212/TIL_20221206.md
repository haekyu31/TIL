# TIL  Linux


```bash
less # 파일을 스크롤 하면서 읽을 수 있는 명령어
f # 한 화면 아래로 
b # 한 화면 위로
j # 한 행 아래로
k # 한 행 위로 
q # less 종료

/ # less 입력된 상태에서 검색하고 싶은 문자열을 입력
/ # 문자열 아래 방향으로 검색
? # 문자열 위 방향으로 검색
n # 다음 검색 결과로 이동
N # 이전 검색 결과로 이동 

cp # 파일 복사 
cp file1 dir1 # 파일을 특정 디렉터리 안에 복사
cp file1 file2 file3 dir2 # dir2에 파일 세 개 복사하기
cp file1 file2 # file1을 file2 이름으로 복사하기 file2의 내용이 없어지고 file1의 내용이 복사됨
cp -i file1 file2 # 덮어쓸지 여부를 확인
cp dir2 dir3 # 디렉터리 복사하려는 경우 -r 옵션을 지정하여 모두 재귀적으로 복사

mv # 파일 위치를 옮길 때 파일 이름을 바꿀 때

ln # 링크 하드링크란 한 파일 원본에 이름을 여러개 붙이는 기능 삭제해도 하드 링크로 연결된 나머지 파일들은 남아있다.
ln -s # 심볼릭 링크 확인 가능 화살표로 표시
ls -l # 심볼릭 링크란 원본 파일에 대한 정보가 담긴 작은 특수 파일이며 하드 링크와 달리 원본과 구별된다.
# 심볼릭 링크를 지우지 않은 채 원본 파일 삭제 가능 그러면 심볼릭 링크가 깨진 상태가 된다. 깨진 파일을 실행하면 오류
# 링크가 유용한 경우

# 긴 경로 명을 짧게
ln -s local/work/code/project/source source
cd source

# 여러 버전의 프로그램이 있는 경우
ls -l



```
