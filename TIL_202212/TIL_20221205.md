# TIL linux


```bash
command line 
echo # 출력
ctrl + b, f # 이전으로, 앞으로 이동
date # 현재시간 출력
exit # 로그아웃
ctrl + a, e # 맨앞으로 , 맨 뒤로 이동
alt + b, f # 한단어 뒤로, 앞으로 이동
ctrl + h # 뒤 한문자 삭제
ctrl + d # 한 문자 삭제
ctrl + k # 커서 위치에서 끝까지 삭제
ctrl + u # 첫 문자까지 삭제
ctrl + y # 마지막 지운 내용 삽입
ctrl + s # 화면 표시 잠금
ctrl + q # 화면 표시 잠금 해제
ctrl + c # 명령 강제 종료
ctrl + l # 화면 삭제 clear

Tab # 자동완성

ctrl + p # 이전 명령 이동
ctrl + n # 다음 명령 이동
ctrl + r # 이력 검색


pwd # 현재 디렉터리 출력
cd # 현재 디렉터리 변경
ls # 디렉터리 안의 파일을 출력
cd ~ # 홈디렉터리로 이동
ls / /usr # 여러 디렉터리를 인자로 지정하여 ls 명령어 실행

* # 임의의 문자열
? # 임의의 한 문자 
-l # 커맨드 라인 옵션 파일 속성 상세저보까지 출력
-F # 파일의 종류를 의미하는 기호 추가

mkdir # 디렉터리 생성
-p # 중간경로 디렉터리 생성 옵션
touch # 파일 생성
rm # 파일 제거
-r # 디렉터리 삭제 대상이 되는 디렉터리와 안의 파일을 전부 삭제
rm -i # 파일 삭제전 확인 메시지 출력
rmdir # 빈디렉터리 삭제 비어있지 않다면 에러메시지 출력
.file # 숨김파일

cat # 파일 내용 출력하기    
-n # 행번호 표시하기
cat # 키보드 입력을 그대로 출력하기
ctrl + d # cat 명령어 종료

```
